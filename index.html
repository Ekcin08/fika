/* ========================================================= FIKA-SCRIPT (Stockholm-tid) + Check-in 15:00–15:15 - NEJ + nedräkning - JA 15:00–15:15 - Checka in knapp 15:00–15:15 - Efter check-in => fika tills 16:00 - Skapar UI automatiskt om det saknas i HTML ========================================================= */ document.addEventListener("DOMContentLoaded", () => { // ---------- 1) Hämta/Skapa UI-element ---------- const ui = ensureUI(); // ---------- 2) Tid i Stockholm (med fallback) ---------- function getSwedishDate() { // Din originalmetod, men med fallback om den skulle bli "Invalid Date" const d = new Date( new Date().toLocaleString("en-US", { timeZone: "Europe/Stockholm" }) ); return isNaN(d.getTime()) ? new Date() : d; } // ---------- 3) Namnsdag (valfri) ---------- async function loadNameday() { const now = getSwedishDate(); const y = now.getFullYear(); const m = String(now.getMonth() + 1).padStart(2, "0"); const d = String(now.getDate()).padStart(2, "0"); try { const r = await fetch(`https://api.dryg.net/dagar/v2.1/${y}/${m}/${d}`); const data = await r.json(); const names = data?.dagar?.[0]?.namnsdag ?? []; ui.nameday.textContent = names.length ? `Namnsdag: ${names.join(", ")}` : "Namnsdag saknas"; } catch (e) { ui.nameday.textContent = "Namnsdag saknas"; } } // ---------- 4) Check-in (sparas per dag) ---------- function todayKey(now) { return `${now.getFullYear()}-${String(now.getMonth() + 1).padStart(2, "0")}-${String(now.getDate()).padStart(2, "0")}`; } function getCheckedIn(now) { const key = `fika_checkedin_${todayKey(now)}`; return localStorage.getItem(key) === "1"; } function setCheckedIn(now, value) { const key = `fika_checkedin_${todayKey(now)}`; localStorage.setItem(key, value ? "1" : "0"); } ui.checkinBtn.addEventListener("click", () => { const now = getSwedishDate(); // Bara tillåtet under checkin-fönstret if (isWithinCheckinWindow(now)) { setCheckedIn(now, true); render(); // uppdatera direkt } }); // ---------- 5) Tidsfönster ---------- function atTime(now, h, m, s = 0, ms = 0) { const t = new Date(now); t.setHours(h, m, s, ms); return t; } function isWeekend(now) { const day = now.getDay(); // 0=sön, 6=lör return day === 0 || day === 6; } function isWithinCheckinWindow(now) { const start = atTime(now, 15, 0, 0, 0); const end = atTime(now, 15, 15, 0, 0); return now >= start && now < end; } // ---------- 6) Render-logik (hjärtat) ---------- function render() { const now = getSwedishDate(); // Grundtider const fikaStart = atTime(now, 15, 0, 0, 0); const fikaEnd = atTime(now, 15, 15, 0, 0); const checkedIn = getCheckedIn(now); // Om checkad in: fika gäller till 16:00 const fikaExtendedEnd = atTime(now, 16, 0, 0, 0); // Fredags-"helg" logik: // - Om man INTE checkat in: helg efter 15:15 // - Om man checkat in: helg efter 16:00 (för att fika pågår längre) const friday = now.getDay() === 5; const fridayWeekendStart = checkedIn ? fikaExtendedEnd : fikaEnd; const fridayAfter = friday && now >= fridayWeekendStart; // Reset UI ui.result.classList.remove("pulse"); ui.countdown.textContent = ""; ui.checkinWrap.style.display = "none"; ui.checkinStatus.textContent = ""; // HELG if (isWeekend(now) || fridayAfter) { ui.result.textContent = "NU ÄR DET HELG 🎉"; return; } // Om checkad in och klockan är före 16:00 => FIKA if (checkedIn && now < fikaExtendedEnd) { ui.result.textContent = "JA ☕"; ui.checkinStatus.textContent = "Du har checkat in till fikat ✅ (fika till 16:00)"; return; } // Vanligt fika-fönster 15:00–15:15 if (now >= fikaStart && now < fikaEnd) { ui.result.textContent = "JA ☕"; // Visa checkin-knappen bara om man inte checkat in än if (!checkedIn) { ui.checkinWrap.style.display = "block"; ui.checkinStatus.textContent = "Checka in till fikat för att förlänga fika till 16:00 👇"; } return; } // SNART (14:55–15:00) const soonStart = new Date(fikaStart.getTime() - 5 * 60 * 1000); if (now >= soonStart && now < fikaStart) { ui.result.textContent = "SNART ☕"; ui.result.classList.add("pulse"); // Countdown till 15:00 ui.countdown.textContent = formatCountdown(fikaStart - now); return; } // Efter 15:15 (och inte checkad in): “missade”-fönster kan du ha kvar om du vill // Här behåller vi din idé men gör den enkel: const missedStart = atTime(now, 15, 30, 0, 0); const missedEnd = atTime(now, 15, 45, 0, 0); if (now >= missedStart && now < missedEnd) { ui.result.textContent = "DU MISSADE FIKAT"; return; } // Annars: NEJ + nedräkning till nästa fika (idag 15:00 om före, annars imorgon 15:00) ui.result.textContent = "NEJ"; let target = new Date(fikaStart); if (now >= fikaEnd) { target.setDate(target.getDate() + 1); target.setHours(15, 0, 0, 0); } ui.countdown.textContent = formatCountdown(target - now); } // ---------- 7) Format countdown ---------- function formatCountdown(ms) { const diff = Math.max(0, ms); const totalSecs = Math.floor(diff / 1000); const hours = Math.floor(totalSecs / 3600); const mins = Math.floor((totalSecs % 3600) / 60); const secs = totalSecs % 60; if (hours > 0) return `Nästa fika om ${hours}h ${mins}m ${secs}s`; return `Nästa fika om ${mins}m ${secs}s`; } // ---------- 8) Skapa UI om det saknas ---------- function ensureUI() { // Försök hitta befintliga element let result = document.getElementById("result"); let countdown = document.getElementById("countdown"); let nameday = document.getElementById("nameday"); let checkinWrap = document.getElementById("checkinWrap"); let checkinBtn = document.getElementById("checkinBtn"); let checkinStatus = document.getElementById("checkinStatus"); // Om något saknas: skapa en enkel container längst upp i body if (!result || !countdown || !nameday || !checkinWrap || !checkinBtn || !checkinStatus) { let root = document.getElementById("fikaWidget"); if (!root) { root = document.createElement("div"); root.id = "fikaWidget"; root.style.maxWidth = "520px"; root.style.margin = "20px auto"; root.style.padding = "16px"; root.style.border = "1px solid rgba(0,0,0,.15)"; root.style.borderRadius = "12px"; root.style.fontFamily = "system-ui, Segoe UI, Arial"; root.style.textAlign = "center"; root.style.background = "rgba(255,255,255,.85)"; document.body.prepend(root); } // Skapa element om de saknas if (!result) { result = document.createElement("div"); result.id = "result"; result.style.fontSize = "48px"; result.style.fontWeight = "800"; result.style.marginBottom = "10px"; root.appendChild(result); } if (!countdown) { countdown = document.createElement("div"); countdown.id = "countdown"; countdown.style.fontSize = "18px"; countdown.style.opacity = "0.85"; countdown.style.marginBottom = "12px"; root.appendChild(countdown); } if (!nameday) { nameday = document.createElement("div"); nameday.id = "nameday"; nameday.style.fontSize = "14px"; nameday.style.opacity = "0.7"; nameday.style.marginBottom = "12px"; root.appendChild(nameday); } if (!checkinWrap) { checkinWrap = document.createElement("div"); checkinWrap.id = "checkinWrap"; checkinWrap.style.marginTop = "10px"; root.appendChild(checkinWrap); } if (!checkinStatus) { checkinStatus = document.createElement("div"); checkinStatus.id = "checkinStatus"; checkinStatus.style.fontSize = "14px"; checkinStatus.style.marginBottom = "8px"; checkinStatus.style.opacity = "0.85"; checkinWrap.appendChild(checkinStatus); } if (!checkinBtn) { checkinBtn = document.createElement("button"); checkinBtn.id = "checkinBtn"; checkinBtn.textContent = "Checka in till fika"; checkinBtn.style.padding = "10px 14px"; checkinBtn.style.borderRadius = "10px"; checkinBtn.style.border = "1px solid rgba(0,0,0,.2)"; checkinBtn.style.cursor = "pointer"; checkinBtn.style.fontWeight = "700"; checkinWrap.appendChild(checkinBtn); } // Lite enkel “pulse”-klass utan CSS-fil const styleId = "fikaWidgetStyle"; if (!document.getElementById(styleId)) { const st = document.createElement("style"); st.id = styleId; st.textContent = ` .pulse { animation: pulse 1s infinite; } @keyframes pulse { 0% { transform: scale(1); } 50% { transform: scale(1.03); } 100% { transform: scale(1); } } `; document.head.appendChild(st); } } return { result, countdown, nameday, checkinWrap, checkinBtn, checkinStatus }; } // ---------- Start ---------- loadNameday(); render(); setInterval(render, 1000); });
